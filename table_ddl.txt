CREATE TABLE `user` (
  `id` int(11) unsigned NOT NULL AUTO_INCREMENT COMMENT 'ID',
  `b_id` int(11) DEFAULT NULL,
  `name` varchar(50) DEFAULT '' COMMENT '姓名',
  `telephone` varchar(20) DEFAULT '' COMMENT '电话',
  `role_id` int(11) DEFAULT NULL COMMENT '角色ID',
  `photo` varchar(255) DEFAULT '' COMMENT '头像',
  `wechat_id` varchar(50) NOT NULL DEFAULT '' COMMENT '微信ID',
  `deleted` tinyint(1) NOT NULL DEFAULT '0' COMMENT '是否删除',
  `password` varchar(255) DEFAULT '' COMMENT '加密密码',
  `create_time` timestamp NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `user_status` tinyint(1) DEFAULT NULL COMMENT '1、待审核2、已通过3、已拒绝',
  `content` varchar(255) DEFAULT NULL,
  `department_id` int(11) DEFAULT NULL COMMENT '部门ID',
  `position_id` int(11) DEFAULT NULL COMMENT '职务ID',
  `area_id` int(11) DEFAULT NULL COMMENT '区域ID',
  `instance_id` int(11) DEFAULT NULL,
  `instance_name` varchar(100) DEFAULT NULL,
  `receive_msg` tinyint(1) DEFAULT '1' COMMENT '是否接收消息：0不接收，1接收',
  `web_key` varchar(8) DEFAULT NULL COMMENT 'web接口签名key',
  `web_key_time` int(11) DEFAULT NULL COMMENT 'web接口签名key到期时间',
  `app_key` varchar(8) DEFAULT NULL COMMENT 'app接口签名key',
  `app_key_time` int(11) DEFAULT NULL COMMENT 'app接口签名key到期时间',
  `user_type` tinyint(1) DEFAULT '3' COMMENT '用户类型 1超级管理员 2普通管理员 3业务员',
  `is_online` tinyint(4) NOT NULL DEFAULT '0' COMMENT '用户是否在线(0=>不在,1=>在)',
  `online_time` datetime DEFAULT NULL COMMENT '在线的查询时间',
  `is_app` tinyint(4) NOT NULL DEFAULT '0' COMMENT '是否是app注册',
  `is_login` tinyint(4) NOT NULL DEFAULT '0' COMMENT '是否登录',
  `phone_brand` varchar(50) DEFAULT NULL,
  `phone_model` varchar(50) DEFAULT NULL,
  `phone_os` varchar(50) DEFAULT NULL,
  `app_version` varchar(10) DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=3078 DEFAULT CHARSET=utf8;
